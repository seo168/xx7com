  function getLast3Dates() {

    const today = new Date();

    const format = (d) => {

      const mm = String(d.getMonth() + 1).padStart(2, '0');

      return `${mm}/${dd}`;

    };


    const d1 = new Date(today);

    d1.setDate(today.getDate() - 2);


    const d2 = new Date(today);

    d2.setDate(today.getDate() - 1);


    const d3 = new Date(today);


    return [format(d1), format(d2), format(d3)];

  }

  function getScrollbarWidth(el) {

  }


  function syncPerfHeaderWidths() {

    document.querySelectorAll('.perf-section-wrapper .table-container').forEach(container => {

      const headerTable = container.querySelector(':scope > table.perf-table');

      const scrollArea = container.querySelector('.perf-scroll-area');

      const bodyTable = scrollArea ? scrollArea.querySelector('table.perf-table') : null;


      if (!headerTable || !scrollArea || !bodyTable) return;


      headerTable.classList.add('perf-header-table');

      bodyTable.classList.add('perf-body-table');


      const scrollbarWidth = getScrollbarWidth(scrollArea);

      headerTable.style.width = `calc(100% - ${scrollbarWidth}px)`;

      bodyTable.style.width = `100%`;


      const headCells = headerTable.querySelectorAll('thead th');

      const firstRow = bodyTable.querySelector('tbody tr');


      if (!firstRow) return;


      const widths = Array.from(headCells).map(th => th.getBoundingClientRect().width);


      headCells.forEach((th, i) => {

        th.style.width = `${widths[i]}px`;

      });


      bodyTable.querySelectorAll('tbody tr').forEach(row => {

        const rowCells = row.querySelectorAll('td');

        rowCells.forEach((td, i) => {

          if (widths[i]) td.style.width = `${widths[i]}px`;

        });

      });

    });

  }


  function bindSmartTips() {

    document.querySelectorAll('.smart-tip').forEach(el => {

      el.removeEventListener('mouseenter', handleTipEnter);

      el.removeEventListener('mousemove', handleTipMove);

      el.removeEventListener('mouseleave', handleTipLeave);

      el.addEventListener('mouseenter', handleTipEnter);

      el.addEventListener('mousemove', handleTipMove);

      el.addEventListener('mouseleave', handleTipLeave);

    });

  }


  function handleTipEnter(e) {

    positionTooltip(e.currentTarget);

  }


  function handleTipMove(e) {

    positionTooltip(e.currentTarget);

  }


  function handleTipLeave(e) {

    e.currentTarget.style.removeProperty('--tip-left');

    e.currentTarget.style.removeProperty('--tip-top');

  }


  function updateVisibleTooltips() {

    document.querySelectorAll('.smart-tip:hover').forEach(el => positionTooltip(el));

  }


  function positionTooltip(el) {

    const rect = el.getBoundingClientRect();

    const margin = 14;

    const tooltipWidth = 320;

    const tooltipHeight = 56;


    let left = rect.right + 12;

    let top = rect.top + (rect.height / 2);


    if (left + tooltipWidth > window.innerWidth - margin) {

      left = Math.max(margin, rect.left - tooltipWidth - 12);

    }


    if (top + (tooltipHeight / 2) > window.innerHeight - margin) {

      top = window.innerHeight - margin - (tooltipHeight / 2);

    }


    if (top - (tooltipHeight / 2) < margin) {

      top = margin + (tooltipHeight / 2);

    }


    el.style.setProperty('--tip-left', `${left}px`);

    el.style.setProperty('--tip-top', `${top}px`);

  }

  function getStatModalSub(chartId) {

    const map = {

      cFN: 'Finance Staff Breakdown',

  function getStatMembers(chartId, label) {

    return filtered.filter(r => {

      const b = r[1] || '';

      const j = r[9] || '';

      if (chartId === 'cWD') return (b.includes('出款') || j.includes('出款')) && b === label;

      if (chartId === 'cCS') return (b.includes('客服') || j.includes('客服') || b.includes('风控')) && b === label;

      if (chartId === 'cFN') return (b.includes('财务') || j.includes('财务')) && b === label;

      if (chartId === 'cGEO') return (r[8] || '') === label;

      if (chartId === 'cSH') return (r[5] || '') === label;

      return false;

    }).map(r => ({

      platform: r[0] || '-',

      group: r[1] || '-',

      team: r[9] || '-',

      id: r[2] || '-',

      name: r[3] || '-',

      shift: r[5] || '-',

      country: r[8] || '-'

    }));

  }


  function getTopValue(arr) {

    const map = {};

    arr.filter(Boolean).forEach(v => map[v] = (map[v] || 0) + 1);

    const sorted = Object.entries(map).sort((a,b) => b[1] - a[1]);

    return sorted.length ? sorted[0][0] : '';

  }


  function hexToRgba(hex, alpha) {

    const cleanHex = hex.replace('#', '');

    const bigint = parseInt(cleanHex.length === 3

      ? cleanHex.split('').map(c => c + c).join('')

      : cleanHex, 16);

    const r = (bigint >> 16) & 255;

    const g = (bigint >> 8) & 255;

    const b = bigint & 255;

    return `rgba(${r}, ${g}, ${b}, ${alpha})`;

  }
